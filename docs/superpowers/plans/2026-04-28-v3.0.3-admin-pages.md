```typescript
type CsvCell = string | number | boolean | Date | null | undefined;

const needsQuoting = (s: string): boolean =>
  s.includes(',') || s.includes('"') || s.includes('\n') || s.includes('\r');

const escapeCell = (value: CsvCell): string => {
  if (value === null || value === undefined) return '';
  if (value instanceof Date) {
    return value.toISOString().slice(0, 10); // YYYY-MM-DD
  }
  const s = String(value);
  if (needsQuoting(s)) {
    return `"${s.replace(/"/g, '""')}"`;
  }
  return s;
};

export function toCsv(headers: string[], rows: CsvCell[][]): string {
  const BOM = '﻿';
  const lines = [headers.map(escapeCell).join(',')];
  for (const row of rows) {
    lines.push(row.map(escapeCell).join(','));
  }
  return BOM + lines.join('\n') + '\n';
}

export function csvResponse(filename: string, body: string): Response {
  return new Response(body, {
    headers: {
      'Content-Type': 'text/csv; charset=utf-8',
      'Content-Disposition': `attachment; filename="${filename}"`,
    },
  });
}
```

```typescript
import type { OverdueLoan, OverdueFilters, OverdueBucket } from './types';

const bucketDayRange = (bucket: OverdueBucket | undefined): { min: number; max: number } => {
  switch (bucket) {
    case '1-7':  return { min: 1, max: 7 };
    case '8-30': return { min: 8, max: 30 };
    case '30+':  return { min: 31, max: 99999 };
    case 'all':
    default:     return { min: 1, max: 99999 };
  }
};

export async function fetchOverdueLoans(filters: OverdueFilters = {}): Promise<OverdueLoan[]> {
  const supabase = getSupabaseServerClient();
  const nowIso = new Date().toISOString();

  const { data, error } = await supabase
    .from('Loans')
    .select(`
      id, borrowed_at, due_at, last_reminded_at,
      copy:Copies(id, barcode, book:Books(id, title, author, cover_image_url)),
      borrower:Users!Loans_user_id_fkey(id, email, profile:UserProfile(display_name, student_id)),
      reminder_user:Users!Loans_last_reminded_by_fkey(id, email, profile:UserProfile(display_name))
    `)
    .is('returned_at', null)
    .lt('due_at', nowIso)
    .order('due_at', { ascending: true });

  if (error) throw error;

  const range = bucketDayRange(filters.bucket);
  const search = filters.search?.trim().toLowerCase();
  const now = Date.now();

  const all = (data ?? []).map((row: any): OverdueLoan => {
    const dueMs = new Date(row.due_at).getTime();
    const daysOverdue = Math.max(0, Math.floor((now - dueMs) / 86400000));
    const reminderProfile = Array.isArray(row.reminder_user?.profile)
      ? row.reminder_user.profile[0]
      : row.reminder_user?.profile;
    const borrowerProfile = Array.isArray(row.borrower?.profile)
      ? row.borrower.profile[0]
      : row.borrower?.profile;
    const copyRow = Array.isArray(row.copy) ? row.copy[0] : row.copy;
    const bookRow = copyRow && (Array.isArray(copyRow.book) ? copyRow.book[0] : copyRow.book);

    return {
      id: row.id,
      borrowedAt: row.borrowed_at,
      dueAt: row.due_at,
      daysOverdue,
      lastRemindedAt: row.last_reminded_at ?? null,
      lastRemindedByName: reminderProfile?.display_name ?? row.reminder_user?.email ?? null,
      copy: copyRow ? { id: copyRow.id, barcode: copyRow.barcode } : null,
      book: bookRow
        ? { id: bookRow.id, title: bookRow.title, author: bookRow.author ?? null, coverImageUrl: bookRow.cover_image_url ?? null }
        : null,
      borrower: row.borrower
        ? {
            id: row.borrower.id,
            displayName: borrowerProfile?.display_name ?? null,
            studentId: borrowerProfile?.student_id ?? null,
            email: row.borrower.email ?? null,
          }
        : null,
    };
  });

  return all.filter((loan) => {
    if (loan.daysOverdue < range.min || loan.daysOverdue > range.max) return false;
    if (!search) return true;
    const haystack = [
      loan.book?.title, loan.book?.author,
      loan.borrower?.displayName, loan.borrower?.studentId, loan.borrower?.email,
      loan.copy?.barcode,
    ].filter(Boolean).join(' ').toLowerCase();
    return haystack.includes(search);
  });
}
```

```typescript
import type { HistoryFilters, HistoryLoan, HistoryPage, LoanStatus } from './types';

const PAGE_SIZE = 50;

const rangeStartDate = (range: HistoryFilters['range'], custom?: string): Date | null => {
  const now = new Date();
  switch (range) {
    case '30d':       return new Date(now.getTime() - 30 * 86400000);
    case '6m':        return new Date(now.setMonth(now.getMonth() - 6));
    case 'semester':  return new Date(now.setMonth(now.getMonth() - 4));
    case 'custom':    return custom ? new Date(custom) : null;
    case 'all':
    default:          return null;
  }
};

const computeStatus = (returnedAt: string | null, dueAt: string, nowMs: number): LoanStatus => {
  if (returnedAt) return 'returned';
  if (new Date(dueAt).getTime() < nowMs) return 'overdue';
  return 'active';
};

export async function fetchAllLoansHistory(
  filters: HistoryFilters = {},
  page = 1,
): Promise<HistoryPage> {
  const supabase = getSupabaseServerClient();

  let query = supabase
    .from('Loans')
    .select(`
      id, borrowed_at, due_at, returned_at,
      copy:Copies(id, barcode, book:Books(id, title, author, cover_image_url)),
      borrower:Users!Loans_user_id_fkey(id, email, profile:UserProfile(display_name, student_id)),
      handler:Users!Loans_handled_by_fkey(id, email, profile:UserProfile(display_name))
    `, { count: 'exact' })
    .order('borrowed_at', { ascending: false });

  // Range filter
  const startDate = rangeStartDate(filters.range, filters.rangeStart);
  if (startDate) {
    query = query.gte('borrowed_at', startDate.toISOString());
  }
  if (filters.range === 'custom' && filters.rangeEnd) {
    query = query.lte('borrowed_at', filters.rangeEnd);
  }

  // Status filter — done at app layer (see below) because 'overdue' is computed.
  // For pure 'returned' or 'active', narrow at DB level for efficiency:
  const nowIso = new Date().toISOString();
  if (filters.status === 'returned') {
    query = query.not('returned_at', 'is', null);
  } else if (filters.status === 'active') {
    query = query.is('returned_at', null).gte('due_at', nowIso);
  } else if (filters.status === 'overdue') {
    query = query.is('returned_at', null).lt('due_at', nowIso);
  }

  // Pagination
  const from = (page - 1) * PAGE_SIZE;
  const to = from + PAGE_SIZE - 1;
  query = query.range(from, to);

  const { data, count, error } = await query;
  if (error) throw error;

  const nowMs = Date.now();
  const search = (s?: string) => s?.trim().toLowerCase();
  const borrowerQ = search(filters.borrowerQ);
  const bookQ = search(filters.bookQ);
  const handlerQ = search(filters.handlerQ);

  let rows: HistoryLoan[] = (data ?? []).map((row: any): HistoryLoan => {
    const copyRow = Array.isArray(row.copy) ? row.copy[0] : row.copy;
    const bookRow = copyRow && (Array.isArray(copyRow.book) ? copyRow.book[0] : copyRow.book);
    const borrowerProfile = Array.isArray(row.borrower?.profile) ? row.borrower.profile[0] : row.borrower?.profile;
    const handlerProfile = Array.isArray(row.handler?.profile) ? row.handler.profile[0] : row.handler?.profile;
    const borrowedMs = new Date(row.borrowed_at).getTime();
    const endMs = row.returned_at ? new Date(row.returned_at).getTime() : nowMs;
    return {
      id: row.id,
      borrowedAt: row.borrowed_at,
      dueAt: row.due_at,
      returnedAt: row.returned_at ?? null,
      durationDays: Math.max(0, Math.floor((endMs - borrowedMs) / 86400000)),
      status: computeStatus(row.returned_at ?? null, row.due_at, nowMs),
      copy: copyRow ? { id: copyRow.id, barcode: copyRow.barcode } : null,
      book: bookRow ? { id: bookRow.id, title: bookRow.title, author: bookRow.author ?? null, coverImageUrl: bookRow.cover_image_url ?? null } : null,
      borrower: row.borrower
        ? { id: row.borrower.id, displayName: borrowerProfile?.display_name ?? null, studentId: borrowerProfile?.student_id ?? null }
        : null,
      handler: row.handler
        ? { id: row.handler.id, displayName: handlerProfile?.display_name ?? row.handler.email ?? null, isSelfCheckout: row.handler.id === row.borrower?.id }
        : null,
    };
  });

  // App-layer text filters
  if (borrowerQ) rows = rows.filter((r) => `${r.borrower?.displayName ?? ''} ${r.borrower?.studentId ?? ''}`.toLowerCase().includes(borrowerQ));
  if (bookQ) rows = rows.filter((r) => `${r.book?.title ?? ''} ${r.book?.author ?? ''} ${r.copy?.barcode ?? ''}`.toLowerCase().includes(bookQ));
  if (handlerQ) rows = rows.filter((r) => (r.handler?.displayName ?? '').toLowerCase().includes(handlerQ));

  return {
    rows,
    total: count ?? 0,
    active: rows.filter((r) => r.status === 'active').length,
    returned: rows.filter((r) => r.status === 'returned').length,
    overdue: rows.filter((r) => r.status === 'overdue').length,
    page,
    pageSize: PAGE_SIZE,
  };
}
```

```typescript
const PREFIX = 'SWI-';
const PAD = 5;

export function computeNextBarcodes(existing: string[], count: number): string[] {
  if (count <= 0) throw new Error('count must be >= 1');
  if (count > 20) throw new Error('count must be <= 20');

  let max = 0;
  for (const code of existing) {
    if (!code?.startsWith(PREFIX)) continue;
    const tail = code.slice(PREFIX.length);
    if (!/^\d+$/.test(tail)) continue;
    const num = parseInt(tail, 10);
    if (num > max) max = num;
  }

  const result: string[] = [];
  for (let i = 1; i <= count; i++) {
    const num = max + i;
    result.push(PREFIX + num.toString().padStart(PAD, '0'));
  }
  return result;
}
```

```typescript
  id: string; title: string; author: string | null; copyCount: number;
} | null> {
  const supabase = getSupabaseServerClient();
  const cleaned = isbn.replace(/[^0-9X]/gi, '');
  if (!cleaned) return null;

  const { data, error } = await supabase
    .from('Books')
    .select('id, title, author, copies:Copies(id)')
    .eq('isbn', cleaned)
    .maybeSingle();

  if (error) {
    console.error('[findBookByIsbn]', error);
    throw error;
  }
  if (!data) return null;

  const copies = Array.isArray(data.copies) ? data.copies : [];
  return {
    id: data.id,
    title: data.title,
    author: data.author ?? null,
    copyCount: copies.length,
  };
}

export async function getNextAvailableBarcodes(count: number): Promise<string[]> {
  const supabase = getSupabaseServerClient();

  // Pull only SWI- codes, sorted desc by string — the leading 0-padding
  // makes this lexicographically equal to numeric for ≤5 digits.
  // We still parse defensively in computeNextBarcodes.
  const { data, error } = await supabase
    .from('Copies')
    .select('barcode')
    .like('barcode', 'SWI-%')
    .order('barcode', { ascending: false })
    .limit(50);

  if (error) throw error;
  const existing = (data ?? []).map((r: any) => r.barcode).filter(Boolean) as string[];

  // Reuse the pure helper
  const { computeNextBarcodes } = await import('@/app/lib/barcode');
  return computeNextBarcodes(existing, count);
}
```

```typescript
import { revalidatePath } from 'next/cache';
import { getDashboardSession } from '@/app/lib/auth/session';
import { getSupabaseServerClient } from '@/app/lib/supabase/server';

const ONE_DAY_MS = 24 * 60 * 60 * 1000;

export type SendReminderResult =
  | { ok: true; lastRemindedAt: string }
  | { ok: false; reason: 'unauthorized' | 'not_overdue' | 'too_recent' | 'failed'; message: string };

export async function sendReminder(loanId: string): Promise<SendReminderResult> {
  const { user } = await getDashboardSession();
  if (!user || (user.role !== 'staff' && user.role !== 'admin')) {
    return { ok: false, reason: 'unauthorized', message: 'Not allowed.' };
  }

  const supabase = getSupabaseServerClient();
  const nowIso = new Date().toISOString();

  // Load loan with current state
  const { data: loan, error: loadError } = await supabase
    .from('Loans')
    .select(`
      id, user_id, due_at, returned_at, last_reminded_at,
      copy:Copies(book:Books(title))
    `)
    .eq('id', loanId)
    .maybeSingle();

  if (loadError || !loan) {
    return { ok: false, reason: 'failed', message: 'Loan not found.' };
  }
  if (loan.returned_at || new Date(loan.due_at).getTime() >= Date.now()) {
    return { ok: false, reason: 'not_overdue', message: 'This loan is no longer overdue.' };
  }
  if (loan.last_reminded_at && Date.now() - new Date(loan.last_reminded_at).getTime() < ONE_DAY_MS) {
    return { ok: false, reason: 'too_recent', message: 'A reminder was already sent within the last 24 hours.' };
  }

  const copyRow = Array.isArray(loan.copy) ? loan.copy[0] : loan.copy;
  const bookRow = copyRow && (Array.isArray(copyRow.book) ? copyRow.book[0] : copyRow.book);
  const title = bookRow?.title ?? 'a library book';
  const daysOverdue = Math.floor((Date.now() - new Date(loan.due_at).getTime()) / ONE_DAY_MS);

  // Insert notification + update loan in two writes (Supabase does not expose tx via JS client)
  const { error: notifError } = await supabase.from('Notifications').insert({
    user_id: loan.user_id,
    title: 'Overdue loan reminder',
    body: `Your loan of "${title}" is ${daysOverdue} day${daysOverdue === 1 ? '' : 's'} overdue. Please return it as soon as possible.`,
    kind: 'overdue_reminder',
  });
  if (notifError) {
    console.error('[sendReminder] notif insert failed', notifError);
    return { ok: false, reason: 'failed', message: 'Could not write notification.' };
  }

  const { error: updateError } = await supabase
    .from('Loans')
    .update({ last_reminded_at: nowIso, last_reminded_by: user.id })
    .eq('id', loanId);
  if (updateError) {
    console.error('[sendReminder] loan update failed', updateError);
    return { ok: false, reason: 'failed', message: 'Could not record reminder.' };
  }

  revalidatePath('/dashboard/admin/overdue');
  return { ok: true, lastRemindedAt: nowIso };
}
```

```typescript
export type CreateBookInput = {
  title: string;
  author: string;
  isbn?: string;
  publisher?: string;
  publicationYear?: string;
  classification?: string;
  coverImageUrl?: string;
  category?: string;
  tags?: string[];

export type CreateBookResult =
  | { ok: true; bookId: string; barcodes: string[] }
  | { ok: false; message: string; field?: string };

export async function createBookWithCopies(input: CreateBookInput): Promise<CreateBookResult> {
  const { user } = await getDashboardSession();
  if (!user || user.role !== 'admin') {
    return { ok: false, message: 'Not allowed.' };
  }

  const title = input.title?.trim();
  const author = input.author?.trim();
  if (!title) return { ok: false, message: 'Title is required.', field: 'title' };
  if (!author) return { ok: false, message: 'Author is required.', field: 'author' };
  if (!Number.isInteger(input.copies) || input.copies < 1 || input.copies > 20) {
    return { ok: false, message: 'Copies must be between 1 and 20.', field: 'copies' };
  }

  const supabase = getSupabaseServerClient();

  // Insert Book
  const bookPayload = {
    title,
    author,
    isbn: input.isbn?.trim() || null,
    publisher: input.publisher?.trim() || null,
    publication_year: input.publicationYear?.trim() || null,
    classification: input.classification?.trim() || null,
    cover_image_url: input.coverImageUrl?.trim() || null,
    category: input.category || null,
  };

  const { data: bookRow, error: bookError } = await supabase
    .from('Books')
    .insert(bookPayload)
    .select('id')
    .single();

  if (bookError || !bookRow) {
    console.error('[createBookWithCopies] book insert failed', bookError);
    return { ok: false, message: bookError?.message ?? 'Failed to create book.' };
  }

  // Compute barcodes + insert copies (with retry-once on collision)
  let attempt = 0;
  let barcodes: string[] = [];
  while (attempt < 2) {
    barcodes = await getNextAvailableBarcodes(input.copies);
    const copiesPayload = barcodes.map((bc) => ({
      book_id: bookRow.id,
      barcode: bc,
      status: 'available',
    }));
    const { error: copiesError } = await supabase.from('Copies').insert(copiesPayload);
    if (!copiesError) break;
    if (attempt === 1 || !copiesError.message?.includes('duplicate')) {
      // Rollback the Book insert if we cannot create copies
      await supabase.from('Books').delete().eq('id', bookRow.id);
      console.error('[createBookWithCopies] copies insert failed', copiesError);
      return { ok: false, message: 'Could not generate barcodes. Try again.' };
    }
    attempt++;
  }

  // Optional tag links
  if (input.tags?.length) {
    const normalized = Array.from(new Set(input.tags.map((t) => t.trim().toLowerCase()).filter(Boolean))).slice(0, 6);
    if (normalized.length) {
      const { data: existingTags } = await supabase.from('BookTags').select('id, name').in('name', normalized);
      const existingByName = new Map((existingTags ?? []).map((t: any) => [t.name, t.id]));
      const toCreate = normalized.filter((n) => !existingByName.has(n));
      if (toCreate.length) {
        const { data: created } = await supabase.from('BookTags').insert(toCreate.map((name) => ({ name }))).select('id, name');
        (created ?? []).forEach((t: any) => existingByName.set(t.name, t.id));
      }
      const linkRows = normalized.map((n) => ({ book_id: bookRow.id, tag_id: existingByName.get(n) })).filter((r) => r.tag_id);
      if (linkRows.length) await supabase.from('BookTagsLinks').insert(linkRows);
    }
  }

  revalidatePath('/dashboard/admin/books/new');
  revalidatePath('/dashboard/book/items');
  return { ok: true, bookId: bookRow.id, barcodes };
}
```

```tsx
                      <p className="truncate font-mono text-[11px] text-swin-charcoal/55 dark:text-white/55">{loan.borrower?.studentId ?? loan.borrower?.email ?? ''}</p>
                    </div>
                  </div>
                </td>
                <td className="px-4 py-3.5 font-mono text-[11px] text-swin-charcoal/60 dark:text-white/60">{new Date(loan.borrowedAt).toLocaleDateString('en-MY')}</td>

                  </span>
                </td>
                <td className="px-4 py-3.5 text-[12px] text-swin-charcoal/60 dark:text-white/60">
                  {loan.lastRemindedAt
                    ? `Reminded ${formatRelative(loan.lastRemindedAt)}${loan.lastRemindedByName ? ` by ${loan.lastRemindedByName}` : ''}`

                </td>
              </tr>
            ))}
            {loans.length === 0 && (
              <tr><td className="px-4 py-8 text-center text-slate-500" colSpan={7}>No overdue loans match these filters.</td></tr>
            )}
          </tbody>
        </table>
      </div>
    </div>
  );

function formatRelative(iso: string): string {
  const diffMs = Date.now() - new Date(iso).getTime();
  const mins = Math.floor(diffMs / 60000);
  if (mins < 60) return `${mins}m ago`;
  const hrs = Math.floor(mins / 60);
  if (hrs < 24) return `${hrs}h ago`;
  const days = Math.floor(hrs / 24);
  return `${days}d ago`;
}
```
